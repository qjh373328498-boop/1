#### 2.2 财务指标计算

```python
# 计算财务比率
def calculate_ratios(df):
    ratios = {}
    
    # 流动比率 = 流动资产 / 流动负债
    ratios['流动比率'] = df['流动资产'] / df['流动负债']
    
    # 资产负债率 = 总负债 / 总资产
    ratios['资产负债率'] = df['总负债'] / df['总资产']
    
    # 净资产收益率 = 净利润 / 净资产
    ratios['ROE'] = df['净利润'] / df['净资产']
    
    # 销售毛利率 = (收入 - 成本) / 收入
    ratios['毛利率'] = (df['营业收入'] - df['营业成本']) / df['营业收入']
    
    return pd.DataFrame(ratios)

# 使用示例
ratios_df = calculate_ratios(financial_df)
```

### 模块 4：统计分析方法

#### 4.1 描述性统计

```python
from scipy import stats

# 基本统计量
df.describe()

# 分位数
df['营业收入'].quantile([0.25, 0.5, 0.75])

# 变异系数（离散程度）
cv = df['营业收入'].std() / df['营业收入'].mean()
```

#### 4.2 假设检验

```python
# t 检验（比较两组均值）
from scipy.stats import ttest_ind

# 比较改革前后利润差异
reform_before = df[df['时期'] == '改革前']['利润']
reform_after = df[df['时期'] == '改革后']['利润']
t_stat, p_value = ttest_ind(reform_before, reform_after)
print(f'p 值：{p_value:.4f}')
if p_value < 0.05:
    print('差异显著')
```

#### 4.3 回归分析

```python
import statsmodels.api as sm

# 线性回归（收入对利润的影响）
X = df['营业收入']
y = df['净利润']
X = sm.add_constant(X)  # 添加截距项
model = sm.OLS(y, X).fit()
print(model.summary())

# 解读：
# - R-squared：拟合优度
# - coef：回归系数
# - P>|t|：显著性（<0.05 为显著）
```

### 项目 3：成本性态分析

**任务**：用高低点法分解混合成本

```python
# 高低点法
def cost_separation(activity, cost):
    # 找到最高点和最低点
    high_idx = activity.idxmax()
    low_idx = activity.idxmin()
    
    # 计算单位变动成本
    b = (cost[high_idx] - cost[low_idx]) / (activity[high_idx] - activity[low_idx])
    
    # 计算固定成本
    a = cost[high_idx] - b * activity[high_idx]
    
    return a, b

# 使用示例
fixed_cost, variable_cost_per_unit = cost_separation(df['产量'], df['总成本'])
```
